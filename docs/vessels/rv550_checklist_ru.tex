\section{Инструкции по взлёту}
\paragraph{Взлёт на ранее выбранную орбиту}~
\begin{itemize}
	\item Нажать \reg{SRB IGNT}
	\item Вычислительный комплекс произведёт автоматическое ведение ракеты по траектории взлёта
	\item Перейти к инструкции \textbf{Выход на орбиту}
\end{itemize}

\paragraph{Взлёт с выбором орбиты}~
\begin{itemize}
  \item Используя клавиши на МФУ выбрать точку входа, наклон орбиты
	\item Нажать \reg{SRB IGNT}
	\item Вычислительный комплекс произведёт автоматическое ведение ракеты по траектории взлёта
	\item Перейти к инструкции \textbf{Выход на орбиту}
\end{itemize}

\paragraph{Взлёт по заданной программе}~
\begin{itemize}
  \item В данной инструкции апогей указан как 456 км, перигей указан как 234 км. Нужно заменить указанные значение на нужные значение апогея и перигея.
  \item Выбрать экран МФЭ \reg{PROGRAM}
  \item Используя клавиши \reg{+} и \reg{-} выбрать команду 1
  \item Ввести команду в контроллер событий: \reg{222 456} (456 км)
  \item Нажать \reg{EXEC}
  \item Нажать \reg{+}
  \item Ввести команду в контроллер событий: \reg{642 234} (234 км)
  \item Нажать \reg{EXEC}
  \item Нажать \reg{SRB IGNT}
  \item Перейти к инструкции \textbf{Автоматические манёвры на орбите}
\end{itemize}

\section{Инструкции по работе на орбите}
\paragraph{Выход на орбиту (ручной)}~
\begin{itemize}
  \item Подождать сброса 2 ступени
	\item Если индикатор кнопки \reg{RCS ON} не горит: нажать кнопку \reg{RCS ON} для включения РСУ
	\item Нажать \reg{ATT PRO}
	\item Убедится, что КА находится в режиме полёта 32 (горит индикатор кнопки \reg{ATT PRO})
	\item Подождать выравнивания по вектору скорости
	\item Убедится, что ручка тяги установлена в 0\%
	\item Нажать кнопку \reg{ORB ENGN} для включения двигателей орбитального маневрирования
	\item Убедится, что индикатор кнопки \reg{ORB ENGN} горит, но не мигает
	\item Если индикатор кнопки \reg{ORB ENGN} мигает, выставить тягу в 0\%
	\item Подождать наступления апогея орбиты, или перейти к следующему шагу если апогей уже пройден
	\item Увеличить тягу до 100\%
	\item Подождать, пока будут достигнуты нужные орбитальные параметры
	\item Выставить тягу в 0\%
	\item Нажать кнопку \reg{ORB ENGN} для отключения двигателей орбитального маневрирования
	\item Убедится, что индикатор кнопки \reg{ORB ENGN} не горит
	
	\item Если требуются дополнительные манёвры перейти к инструкции \textbf{Орбитальные манёвры}
\end{itemize}

\paragraph{Выход на орбиту (автоматический)}~
\begin{itemize}
  \item Подождать сброса 2 ступени
  \item Выбрать экран МФЭ \reg{PROGRAM}
  \item Используя клавиши \reg{+} и \reg{-} выбрать команду 1
  \item Ввести команду в контроллер событий: \reg{222 234} (перигей 234 км)
  \item Нажать \reg{EXEC}
\end{itemize}

\paragraph{Орбитальные манёвры (вручную)}~
\begin{itemize}
  \item Убедится, что ручка тяги установлена в 0\%
	\item Нажать кнопку \reg{ORB ENGN} для включения двигателей орбитального маневрирования
	\item Убедится, что индикатор кнопки \reg{ORB ENGN} горит, но не мигает
	\item Если индикатор кнопки \reg{ORB ENGN} мигает, выставить тягу в 0\%
	\item Увеличить тягу до 100\%
	\item Подождать, пока будут достигнуты нужные орбитальные параметры
	\item Выставить тягу в 0\%
	\item Нажать кнопку \reg{ORB ENGN} для отключения двигателей орбитального маневрирования
	\item Убедится, что индикатор кнопки \reg{ORB ENGN} не горит
\end{itemize}

\paragraph{Орбитальные манёвры (автоматические)}~
\begin{itemize}
  \item Ввести нужную команду в контроллер событий
  \item Нажать \reg{EXEC}
\end{itemize}

\section{Инструкции по входу в атмосферу}
\paragraph{Баллистический спуск (вручную)}~
\begin{itemize}
  \item Выполнить инструкцию \textbf{Орбитальные манёвры} что-бы расположить аппарат на нужной для входа высоте (не ниже 150 км)
  \item Нажать \reg{ATT RET}
  \item Убедится, что аппарат развёрнут против направления движения
  \item Выполнить манёвр схода с орбиты (используя систему орбитального маневрирования) пока угол входа в атмосферу не будет равен -1.0 градуса (см инструкцию \textbf{Орбитальные манёвры (вручную)})
  \item Нажать \reg{DROP ARM}
  \item Нажать \reg{DROP STAGE} для отделения приборно-агрегатного отсека
  \item Нажать \reg{ATT RET} для восстановления режима автопилота
  \item КА выполнит вход в атмосферу автоматически
\end{itemize}

\paragraph{Баллистический спуск (автоматический)}~
\begin{itemize}
  \item Выполнить инструкцию \textbf{Орбитальные манёвры} что-бы расположить аппарат на нужной для входа высоте
  \item Не позднее чем через 5 минут до точки входа ввести команду \reg{431} (выполнить сход с орбиты)
  \item Нажать \reg{EXEC}
  \item Нажать \reg{+}
  \item Ввести команду \reg{3F1 7193} (отделение приборно-агрегатного отсека при входе в атмосферу)
  \item Нажать \reg{EXEC}
  \item КА выполнит вход в атмосферу автоматически
\end{itemize}

\section{Недопустимые режимы работы}
\begin{itemize}
  \item Не входить в атмосферу под углом менее -0.5 градусов
  \item Не входить в атмосферу под углом, превышающим -2.5 градуса
  \item Не спускаться ниже 120 км с не отстыкованым приборно-агрегатным отсеком
  \item Всегда проверять состояние бортового вычислительного комплекса после перезагрузки
  \item Всегда отключать систему орбитального маневрирования после использования
  \item Всегда отстыковывать приборно-агрегатный отсек в точке, или до точки входа в атмосферу (120 км)
  \item Не допускается открытие окон для проветривания при нахождении в космическом пространстве
\end{itemize}
